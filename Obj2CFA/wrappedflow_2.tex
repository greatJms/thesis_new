\begin{lstlisting}[multicols=2, xleftmargin=10pt]
class Container {
  Object elem;
  void add(Object el) {
    this.elem = el;
  }
  Itr itr() {
    Object e = this.elem;
    Itr itr = new Itr(e);//It
    return itr;
  }
}
class Itr {
  Object next;
  Itr(Object obj) {
    this.next = obj;
  }
  Object next() {
    return this.next;
  }
}
void main() {
  Container c1 = new Container();//C1
  c1.add(new A());//A
  Itr i1 = c1.itr();
  object o1 = i1.next();
	
  Container c2 = new Container();//C2
  c2.add(new B());//B
  Itr i2 = c2.itr();
  object o2 = i2.next();
}
\end{lstlisting}
